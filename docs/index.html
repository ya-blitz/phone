API управления телефоном.

Пример использования:

const { createPhone, BEEP_CODES } = require('@yandex-blitz/phone');

const phone = createPhone({ data: '' });
phone.connect()
    .then(() => {
        phone.dial('273145');
    })
    .catch(() => {
        phone.beep(BEEP_CODES.ERROR);
    });
